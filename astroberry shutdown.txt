
#
# Give user 'astroberry' shutdown/reboot privileges
#



# COPY and PASTE this command into terminal

sudo visudo /etc/sudoers.d/010_astroberry-nopasswd



# COPY and PASTE this line into terminal

astroberry ALL=(ALL) NOPASSWD: /sbin/reboot, /sbin/shutdown, /sbin/poweroff



# Save & Exit

Ctrl + O, Enter
Ctrl + X, Enter
